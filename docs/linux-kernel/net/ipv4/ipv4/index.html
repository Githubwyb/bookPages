    一、ipv4收包后如何处理 #

    1. ip层包入口 # 

     1// net/ipv4/ip_input.c
 2/*
 3 * 	Deliver IP Packets to the higher protocol layers.
 4 */
 5int ip_local_deliver(struct sk_buff *skb) {
 6	/*
 7	 *	Reassemble IP fragments.
 8	 */
 9	struct net *net = dev_net(skb->dev);
10
11    // 判断是否为分片包，分片包就先处理分片
12	if (ip_is_fragment(ip_hdr(skb))) {
13		if (ip_defrag(net, skb, IP_DEFRAG_LOCAL_DELIVER))
14			return 0;
15	}
16
17    // 分片处理完毕调用ip_local_deliver_finish()
18	return NF_HOOK(NFPROTO_IPV4, NF_INET_LOCAL_IN,
19		       net, NULL, skb, skb->dev, NULL,
20		       ip_local_deliver_finish);
21}
    • 根据ip包内的协议处理具体的协议
     1// net/ipv4/ip_input.c
 2// ip_local_deliver_finish() -call-> ip_local_deliver_finish()
 3void ip_local_deliver_finish(struct net *net, struct sk_buff *skb, int protocol) {
 4	const struct net_protocol *ipprot;
 5	int raw, ret;
 6
 7resubmit:
 8	raw = raw_local_deliver(skb, protocol);
 9
10    // 这里找到具体的协议的处理函数，然后调用具体的协议处理函数
11	ipprot = rcu_dereference(inet_protos[protocol]);
12	if (ipprot) {
13		if (!ipprot->no_policy) {
14			if (!xfrm4_policy_check(NULL, XFRM_POLICY_IN, skb)) {
15				kfree_skb(skb);
16				return;
17			}
18			nf_reset_ct(skb);
19		}
20        // 相当于调用ipprot->handler(skb)
21		ret = INDIRECT_CALL_2(ipprot->handler, tcp_v4_rcv, udp_rcv,
22				      skb);
23		if (ret < 0) {
24			protocol = -ret;
25			goto resubmit;
26		}
27		__IP_INC_STATS(net, IPSTATS_MIB_INDELIVERS);
28	} else {
29		if (!raw) {
30			if (xfrm4_policy_check(NULL, XFRM_POLICY_IN, skb)) {
31				__IP_INC_STATS(net, IPSTATS_MIB_INUNKNOWNPROTOS);
32                // 这里是服务端会回复一个icmp包，内容就是端口不可达
33				icmp_send(skb, ICMP_DEST_UNREACH,
34					  ICMP_PROT_UNREACH, 0);
35			}
36			kfree_skb(skb);
37		} else {
38			__IP_INC_STATS(net, IPSTATS_MIB_INDELIVERS);
39			consume_skb(skb);
40		}
41	}
42}

    • socket创建的时候会调用pf->create对应ipv4的inet_create
      1// net/ipv4/af_inet.c
  2/*
  3 *	Create an inet socket.
  4 */
  5
  6static int inet_create(struct net *net, struct socket *sock, int protocol,
  7		       int kern)
  8{
  9	struct sock *sk;
 10	struct inet_protosw *answer;
 11	struct inet_sock *inet;
 12	struct proto *answer_prot;
 13	unsigned char answer_flags;
 14	int try_loading_module = 0;
 15	int err;
 16
 17	if (protocol < 0 || protocol >= IPPROTO_MAX)
 18		return -EINVAL;
 19
 20	// 初始化状态为SS_UNCONNETED
 21	sock->state = SS_UNCONNECTED;
 22
 23	/* Look for the requested type/protocol pair. */
 24lookup_protocol:
 25	err = -ESOCKTNOSUPPORT;
 26	rcu_read_lock();
 27	// 从inetsw中找到对应协议的结构体，赋值给answer变量
 28	list_for_each_entry_rcu(answer, &inetsw[sock->type], list) {
 29
 30		err = 0;
 31		/* Check the non-wild match. */
 32		if (protocol == answer->protocol) {
 33			if (protocol != IPPROTO_IP)
 34				break;
 35		} else {
 36			/* Check for the two wild cases. */
 37			if (IPPROTO_IP == protocol) {
 38				protocol = answer->protocol;
 39				break;
 40			}
 41			if (IPPROTO_IP == answer->protocol)
 42				break;
 43		}
 44		err = -EPROTONOSUPPORT;
 45	}
 46
 47	if (unlikely(err)) {
 48		if (try_loading_module < 2) {
 49			rcu_read_unlock();
 50			/*
 51			 * Be more specific, e.g. net-pf-2-proto-132-type-1
 52			 * (net-pf-PF_INET-proto-IPPROTO_SCTP-type-SOCK_STREAM)
 53			 */
 54			if (++try_loading_module == 1)
 55				request_module("net-pf-%d-proto-%d-type-%d",
 56					       PF_INET, protocol, sock->type);
 57			/*
 58			 * Fall back to generic, e.g. net-pf-2-proto-132
 59			 * (net-pf-PF_INET-proto-IPPROTO_SCTP)
 60			 */
 61			else
 62				request_module("net-pf-%d-proto-%d",
 63					       PF_INET, protocol);
 64			goto lookup_protocol;
 65		} else
 66			goto out_rcu_unlock;
 67	}
 68
 69	err = -EPERM;
 70	if (sock->type == SOCK_RAW && !kern &&
 71	    !ns_capable(net->user_ns, CAP_NET_RAW))
 72		goto out_rcu_unlock;
 73
 74	// 将对应协议的操作放到sock里面
 75	sock->ops = answer->ops;
 76	answer_prot = answer->prot;
 77	answer_flags = answer->flags;
 78	rcu_read_unlock();
 79
 80	WARN_ON(!answer_prot->slab);
 81
 82	err = -ENOMEM;
 83	sk = sk_alloc(net, PF_INET, GFP_KERNEL, answer_prot, kern);
 84	if (!sk)
 85		goto out;
 86
 87	err = 0;
 88	if (INET_PROTOSW_REUSE & answer_flags)
 89		sk->sk_reuse = SK_CAN_REUSE;
 90
 91	inet = inet_sk(sk);
 92	inet->is_icsk = (INET_PROTOSW_ICSK & answer_flags) != 0;
 93
 94	inet->nodefrag = 0;
 95
 96	if (SOCK_RAW == sock->type) {
 97		inet->inet_num = protocol;
 98		if (IPPROTO_RAW == protocol)
 99			inet->hdrincl = 1;
100	}
101
102	if (READ_ONCE(net->ipv4.sysctl_ip_no_pmtu_disc))
103		inet->pmtudisc = IP_PMTUDISC_DONT;
104	else
105		inet->pmtudisc = IP_PMTUDISC_WANT;
106
107	inet->inet_id = 0;
108
109	sock_init_data(sock, sk);
110
111	sk->sk_destruct	   = inet_sock_destruct;
112	sk->sk_protocol	   = protocol;
113	sk->sk_backlog_rcv = sk->sk_prot->backlog_rcv;
114
115	inet->uc_ttl	= -1;
116	inet->mc_loop	= 1;
117	inet->mc_ttl	= 1;
118	inet->mc_all	= 1;
119	inet->mc_index	= 0;
120	inet->mc_list	= NULL;
121	inet->rcv_tos	= 0;
122
123	sk_refcnt_debug_inc(sk);
124
125	if (inet->inet_num) {
126		/* It assumes that any protocol which allows
127		 * the user to assign a number at socket
128		 * creation time automatically
129		 * shares.
130		 */
131		inet->inet_sport = htons(inet->inet_num);
132		/* Add to protocol hash chains. */
133		err = sk->sk_prot->hash(sk);
134		if (err) {
135			sk_common_release(sk);
136			goto out;
137		}
138	}
139
140	if (sk->sk_prot->init) {
141		err = sk->sk_prot->init(sk);
142		if (err) {
143			sk_common_release(sk);
144			goto out;
145		}
146	}
147
148	if (!kern) {
149		err = BPF_CGROUP_RUN_PROG_INET_SOCK(sk);
150		if (err) {
151			sk_common_release(sk);
152			goto out;
153		}
154	}
155out:
156	return err;
157out_rcu_unlock:
158	rcu_read_unlock();
159	goto out;
160}
    • inetsw定义，通过inet_register_protosw进行注册
     1// net/ipv4/af_inet.c
 2/* The inetsw table contains everything that inet_create needs to
 3 * build a new socket.
 4 */
 5static struct list_head inetsw[SOCK_MAX];
 6static DEFINE_SPINLOCK(inetsw_lock);
 7...
 8void inet_register_protosw(struct inet_protosw *p)
 9{
10	struct list_head *lh;
11	struct inet_protosw *answer;
12	int protocol = p->protocol;
13	struct list_head *last_perm;
14
15	spin_lock_bh(&inetsw_lock);
16
17	if (p->type >= SOCK_MAX)
18		goto out_illegal;
19
20	/* If we are trying to override a permanent protocol, bail. */
21	last_perm = &inetsw[p->type];
22	list_for_each(lh, &inetsw[p->type]) {
23		answer = list_entry(lh, struct inet_protosw, list);
24		/* Check only the non-wild match. */
25		if ((INET_PROTOSW_PERMANENT & answer->flags) == 0)
26			break;
27		if (protocol == answer->protocol)
28			goto out_permanent;
29		last_perm = lh;
30	}
31
32	/* Add the new entry after the last permanent entry if any, so that
33	 * the new entry does not override a permanent entry when matched with
34	 * a wild-card protocol. But it is allowed to override any existing
35	 * non-permanent entry.  This means that when we remove this entry, the
36	 * system automatically returns to the old behavior.
37	 */
38	list_add_rcu(&p->list, last_perm);
39out:
40	spin_unlock_bh(&inetsw_lock);
41
42	return;
43
44out_permanent:
45	pr_err("Attempt to override permanent protocol %d\n", protocol);
46	goto out;
47
48out_illegal:
49	pr_err("Ignoring attempt to register invalid socket type %d\n",
50	       p->type);
51	goto out;
52}
53EXPORT_SYMBOL(inet_register_protosw);
