    • ipv4里面分析了，当ip层收到驱动的包之后，根据IPPROTO_UDP找到udp_protocol，然后调用handler函数对应udp_rcv
      1// net/ipv4/udp.c
  2/*
  3 *	All we need to do is get the socket, and then do a checksum.
  4 */
  5// udp_rcv -call-> __udp4_lib_rcv
  6int __udp4_lib_rcv(struct sk_buff *skb, struct udp_table *udptable,
  7		   int proto)
  8{
  9	struct sock *sk;
 10	struct udphdr *uh;
 11	unsigned short ulen;
 12	struct rtable *rt = skb_rtable(skb);
 13	__be32 saddr, daddr;
 14	struct net *net = dev_net(skb->dev);
 15	bool refcounted;
 16	int drop_reason;
 17
 18	drop_reason = SKB_DROP_REASON_NOT_SPECIFIED;
 19
 20	/*
 21	 *  Validate the packet.
 22	 */
 23	if (!pskb_may_pull(skb, sizeof(struct udphdr)))
 24		goto drop;		/* No space for header. */
 25
 26	uh   = udp_hdr(skb);
 27	ulen = ntohs(uh->len);
 28	saddr = ip_hdr(skb)->saddr;
 29	daddr = ip_hdr(skb)->daddr;
 30
 31	if (ulen > skb->len)
 32		goto short_packet;
 33
 34	if (proto == IPPROTO_UDP) {
 35		/* UDP validates ulen. */
 36		if (ulen < sizeof(*uh) || pskb_trim_rcsum(skb, ulen))
 37			goto short_packet;
 38		uh = udp_hdr(skb);
 39	}
 40
 41	if (udp4_csum_init(skb, uh, proto))
 42		goto csum_error;
 43
 44	sk = skb_steal_sock(skb, &refcounted);
 45	if (sk) {
 46		struct dst_entry *dst = skb_dst(skb);
 47		int ret;
 48
 49		if (unlikely(rcu_dereference(sk->sk_rx_dst) != dst))
 50			udp_sk_rx_dst_set(sk, dst);
 51
 52		ret = udp_unicast_rcv_skb(sk, skb, uh);
 53		if (refcounted)
 54			sock_put(sk);
 55		return ret;
 56	}
 57
 58    // 这里处理广播和多播场景
 59	if (rt->rt_flags & (RTCF_BROADCAST|RTCF_MULTICAST))
 60		return __udp4_lib_mcast_deliver(net, skb, uh,
 61						saddr, daddr, udptable, proto);
 62
 63    // 根据源地址和目的地址从udptable中找sk结构体
 64	sk = __udp4_lib_lookup_skb(skb, uh->source, uh->dest, udptable);
 65    // 找到直接给对应的sk进行处理
 66	if (sk)
 67		return udp_unicast_rcv_skb(sk, skb, uh);
 68
 69	if (!xfrm4_policy_check(NULL, XFRM_POLICY_IN, skb))
 70		goto drop;
 71	nf_reset_ct(skb);
 72
 73	/* No socket. Drop packet silently, if checksum is wrong */
 74	if (udp_lib_checksum_complete(skb))
 75		goto csum_error;
 76
 77    // 没有socket对应此包，并且checksum是对的，会返回一个icmp目的地址和端口不可达信息
 78	drop_reason = SKB_DROP_REASON_NO_SOCKET;
 79	__UDP_INC_STATS(net, UDP_MIB_NOPORTS, proto == IPPROTO_UDPLITE);
 80	icmp_send(skb, ICMP_DEST_UNREACH, ICMP_PORT_UNREACH, 0);
 81
 82	/*
 83	 * Hmm.  We got an UDP packet to a port to which we
 84	 * don't wanna listen.  Ignore it.
 85	 */
 86	kfree_skb_reason(skb, drop_reason);
 87	return 0;
 88
 89short_packet:
 90	drop_reason = SKB_DROP_REASON_PKT_TOO_SMALL;
 91	net_dbg_ratelimited("UDP%s: short packet: From %pI4:%u %d/%d to %pI4:%u\n",
 92			    proto == IPPROTO_UDPLITE ? "Lite" : "",
 93			    &saddr, ntohs(uh->source),
 94			    ulen, skb->len,
 95			    &daddr, ntohs(uh->dest));
 96	goto drop;
 97
 98csum_error:
 99	/*
100	 * RFC1122: OK.  Discards the bad packet silently (as far as
101	 * the network is concerned, anyway) as per 4.1.3.4 (MUST).
102	 */
103	drop_reason = SKB_DROP_REASON_UDP_CSUM;
104	net_dbg_ratelimited("UDP%s: bad checksum. From %pI4:%u to %pI4:%u ulen %d\n",
105			    proto == IPPROTO_UDPLITE ? "Lite" : "",
106			    &saddr, ntohs(uh->source), &daddr, ntohs(uh->dest),
107			    ulen);
108	__UDP_INC_STATS(net, UDP_MIB_CSUMERRORS, proto == IPPROTO_UDPLITE);
109drop:
110	__UDP_INC_STATS(net, UDP_MIB_INERRORS, proto == IPPROTO_UDPLITE);
111	kfree_skb_reason(skb, drop_reason);
112	return 0;
113}

     1// net/ipv4/udp.c
 2/* wrapper for udp_queue_rcv_skb tacking care of csum conversion and
 3 * return code conversion for ip layer consumption
 4 */
 5static int udp_unicast_rcv_skb(struct sock *sk, struct sk_buff *skb,
 6			       struct udphdr *uh)
 7{
 8	int ret;
 9
10	if (inet_get_convert_csum(sk) && uh->check && !IS_UDPLITE(sk))
11		skb_checksum_try_convert(skb, IPPROTO_UDP, inet_compute_pseudo);
12
13	ret = udp_queue_rcv_skb(sk, skb);
14
15	/* a return value > 0 means to resubmit the input, but
16	 * it wants the return to be -protocol, or 0
17	 */
18	if (ret > 0)
19		return -ret;
20	return 0;
21}

     1// net/ipv4/udp.c
 2static int udp_queue_rcv_skb(struct sock *sk, struct sk_buff *skb)
 3{
 4	struct sk_buff *next, *segs;
 5	int ret;
 6
 7	if (likely(!udp_unexpected_gso(sk, skb)))
 8		return udp_queue_rcv_one_skb(sk, skb);
 9
10	BUILD_BUG_ON(sizeof(struct udp_skb_cb) > SKB_GSO_CB_OFFSET);
11	__skb_push(skb, -skb_mac_offset(skb));
12	segs = udp_rcv_segment(sk, skb, true);
13	skb_list_walk_safe(segs, skb, next) {
14		__skb_pull(skb, skb_transport_offset(skb));
15
16		udp_post_segment_fix_csum(skb);
17		ret = udp_queue_rcv_one_skb(sk, skb);
18		if (ret > 0)
19			ip_protocol_deliver_rcu(dev_net(skb->dev), skb, ret);
20	}
21	return 0;
22}
