    1.3. minisock 缩减版本的socket #

    • 部分场景下不需要sock那么大的结构体，需要缩减一下使用，所以内核定义了两个应对部分场景的结构体
    • 但是在部分处理时（如netfilter的钩子）中，拿到的是struct sock *sk，但是使用sock中的一些值会出问题，因为含义完全不同
    • 但是minisock中均定义了sock_common结构体，在sock_common中的成员含义是一致的，只是部分声明为了union，对不同sock有不同的使用
    • 判断是否为完整sock而非minisock的函数，处理就是判断sk_state，两个场景都有特定的sk_state
    1// include/net/sock.h
2/* This helper checks if a socket is a full socket,
3 * ie _not_ a timewait or request socket.
4 */
5static inline bool sk_fullsock(const struct sock *sk)
6{
7	return (1 << sk->sk_state) & ~(TCPF_TIME_WAIT | TCPF_NEW_SYN_RECV);
8}

    五、bind 绑定地址 #

    • 端口传0会自动分配一个没有占用的端口给socket

    1. 系统调用定义 # 

     1// net/socket.c
 2/*
 3 *	Bind a name to a socket. Nothing much to do here since it's
 4 *	the protocol's responsibility to handle the local address.
 5 *
 6 *	We move the socket address to kernel space before we call
 7 *	the protocol layer (having also checked the address is ok).
 8 */
 9
10int __sys_bind(int fd, struct sockaddr __user *umyaddr, int addrlen)
11{
12	struct socket *sock;
13	struct sockaddr_storage address;
14	int err, fput_needed;
15
16	// fd找socket结构体
17	sock = sockfd_lookup_light(fd, &err, &fput_needed);
18	if (sock) {
19		// 地址拷贝到内核空间
20		err = move_addr_to_kernel(umyaddr, addrlen, &address);
21		if (!err) {
22			err = security_socket_bind(sock,
23						   (struct sockaddr *)&address,
24						   addrlen);
25			if (!err)
26				// 调用真正的bind函数
27				err = sock->ops->bind(sock,
28						      (struct sockaddr *)
29						      &address, addrlen);
30		}
31		fput_light(sock->file, fput_needed);
32	}
33	return err;
34}
35
36SYSCALL_DEFINE3(bind, int, fd, struct sockaddr __user *, umyaddr, int, addrlen)
37{
38	return __sys_bind(fd, umyaddr, addrlen);
39}

    六、listen 监听 #

    1. 系统调用 # 

     1/*
 2 *	Perform a listen. Basically, we allow the protocol to do anything
 3 *	necessary for a listen, and if that works, we mark the socket as
 4 *	ready for listening.
 5 */
 6
 7int __sys_listen(int fd, int backlog)
 8{
 9	struct socket *sock;
10	int err, fput_needed;
11	int somaxconn;
12
13	sock = sockfd_lookup_light(fd, &err, &fput_needed);
14	if (sock) {
15		somaxconn = sock_net(sock->sk)->core.sysctl_somaxconn;
16		if ((unsigned int)backlog > somaxconn)
17			backlog = somaxconn;
18
19		err = security_socket_listen(sock, backlog);
20		if (!err)
21			err = sock->ops->listen(sock, backlog);
22
23		fput_light(sock->file, fput_needed);
24	}
25	return err;
26}
